def to_s(fmt = nil)
  if(fmt)
    #needs work to include distance as well as angle fmt.
    "#{@bearing.strf(fmt)} #{distance.round(4)}m"
  else
    "#{@bearing.strf} #{distance.round(4)}m"
  end
