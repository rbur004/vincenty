def to_s(fmt = nil)
  if fmt
    # needs work to include distance as well as angle fmt.
    return "#{@bearing.strf(fmt)} #{distance.round(4)}m"
  else
    return "#{@bearing.strf} #{distance.round(4)}m"
  end
end
